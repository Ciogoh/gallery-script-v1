let shape;let angleZ = 1.57;let velocityZ = 0;let fsButton; // Variabile per il pulsantefunction preload() {  shape = loadModel('sedia_prova.obj', true, handleSuccess, handleError);}function handleSuccess() { console.log("Modello caricato!"); }function handleError() { console.error("Errore nel caricamento del file .obj"); }function setup() {  let canvas = createCanvas(windowWidth, windowHeight, WEBGL);  canvas.style('display', 'block');    // --- TITOLO ---  let title = createElement('h1', 'INSTABILITË CRONICA');  styleElement(title, '40px', '40px', 'bold', '4px', '1.0');  // --- SOTTOTITOLO ---  let subtitle = createElement('p', 'prova a mantenere in equilibrio la sedia');  styleElement(subtitle, '85px', '14px', 'normal', '1px', '0.5');  // --- TASTO FULLSCREEN ---  fsButton = createButton('SCHERMO INTERO');  fsButton.position(20, 20);  fsButton.mousePressed(toggleFullscreen);    // Stilizzazione pulsante  fsButton.style('background', 'transparent');  fsButton.style('color', 'rgba(255, 255, 255, 0.6)');  fsButton.style('border', '1px solid rgba(255, 255, 255, 0.3)');  fsButton.style('padding', '8px 12px');  fsButton.style('font-family', 'sans-serif');  fsButton.style('font-size', '10px');  fsButton.style('letter-spacing', '2px');  fsButton.style('cursor', 'pointer');  fsButton.mouseOver(() => fsButton.style('background', 'rgba(255, 255, 255, 0.1)'));  fsButton.mouseOut(() => fsButton.style('background', 'transparent'));}// Funzione helper per lo stile dei testifunction styleElement(el, top, size, weight, spacing, opacity) {  el.style('color', `rgba(255, 255, 255, ${opacity})`);  el.style('position', 'fixed');  el.style('top', top);  el.style('left', '50%');  el.style('transform', 'translateX(-50%)');  el.style('margin', '0');  el.style('font-family', 'sans-serif');  el.style('font-size', size);  el.style('font-weight', weight);  el.style('letter-spacing', spacing);  el.style('white-space', 'nowrap');  el.style('pointer-events', 'none');}function toggleFullscreen() {  let fs = fullscreen();  fullscreen(!fs); // Inverte lo stato attuale}function draw() {  background('#131314');  orbitControl();  ambientLight(120);  pointLight(255, 255, 255, 500, -500, 500);  directionalLight(150, 150, 150, -1, 1, -1);  if (shape) {    push();    rotateX(PI);     let targetAngle = map(mouseX, 0, width, 0, HALF_PI);    let force = (targetAngle - angleZ) * 0.05;     let gravity = 0.01;         velocityZ += force + gravity;    velocityZ *= 0.92;     angleZ += velocityZ;    if (angleZ > HALF_PI) {      angleZ = HALF_PI;      velocityZ *= -0.5;    }    rotateZ(angleZ);    noStroke();    specularMaterial(245);    shininess(20);    model(shape);    pop();  }}function windowResized() {  resizeCanvas(windowWidth, windowHeight);}
